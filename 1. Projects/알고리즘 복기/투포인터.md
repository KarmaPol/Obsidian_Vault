import java.util.*;
import java.io.*;

class Solution {
    private Map<String, Integer> gemMap = new HashMap<>();
    
    public int[] solution(String[] gems) {
        int[] answer = {1, gems.length};
        
        // 유니크한 보석 갯수 측정
        Set<String> set = new HashSet<>();
        for(String gem : gems) {
            set.add(gem);
        }
        int uniques = set.size();
        
        // 투포인터
        int en = 0;
        gemMap.put(gems[en], gemMap.getOrDefault(gems[en], 0) + 1);
        for(int st = 0; st < gems.length; st++) {
            while(en < gems.length && !isAll(uniques)) {
                en++;
                if(en != gems.length) gemMap.put(gems[en], gemMap.getOrDefault(gems[en], 0) + 1);
            }
            if(en == gems.length) break;
            
            if(answer[1] - answer[0] > (en+1) - (st+1)){
                answer[0] = st+1; answer[1] = (en+1);
            }
            else if(answer[1] - answer[0] == (en+1) - (st+1) && answer[0] > st+1) {
                answer[0] = st+1; answer[1] = (en+1);
            }
            gemMap.put(gems[st], gemMap.getOrDefault(gems[st], 0) - 1);
            if(gemMap.get(gems[st]) <= 0) gemMap.remove(gems[st]);
        }
        
        return answer;
    }
    
    private boolean isAll(int n) {
        return gemMap.keySet().size() == n;
    }
}
