import java.util.*;
import java.io.*;

class Solution {
    public long solution(int n, int[] times) {
        long answer = 0;
        
        // parametric search - 시간/(심사시간) 의 전체 합 >= n 인 최솟값        
        long left = 0;
        long right = 1000000000000000000L;
        long mid = 0;
        
        long bestTime = right;
        while(left <= right) {
            mid = (left + right)/2;
            System.out.println(mid);
            
            if(isSatisfied(n, times, mid)) {
                bestTime = Math.min(bestTime, mid);
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        
        return bestTime;
    }
    
    private boolean isSatisfied(int n, int[] times, long time) {
        long completed = 0;
        for(int t : times) {
            completed += time/t;
        }
        return n <= completed;
    }
}
